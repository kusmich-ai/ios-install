   - Gold-standard positive-affect scale

## CALCULATION FUNCTIONS

### **Adherence Calculation**

```javascript
function calculateAdherence(stage, dailyLog) {
  // Get last 14 days of logs
  const last14Days = dailyLog.slice(-14);
  
  // Expected rituals per day based on stage
  const ritualsPerDay = {
    1: 2, // HRVB + Awareness Rep
    2: 3, // + Somatic Flow
    3: 4  // + Micro-Action
  }[stage];
  
  const totalExpected = 14 * ritualsPerDay;
  const totalCompleted = last14Days.reduce((sum, day) => {
    return sum + (day.ritualsCompleted?.length || 0);
  }, 0);
  
  return (totalCompleted / totalExpected) * 100;
}
```

### **Delta Calculation**

```javascript
function calculateAverageDelta(weeklyDeltas, baselineScores) {
  const latestWeek = weeklyDeltas[weeklyDeltas.length - 1];
  
  const deltas = {
    regulation: latestWeek.regulation - baselineScores.regulation,
    awareness: latestWeek.awareness - baselineScores.awareness,
    outlook: latestWeek.outlook - baselineScores.outlook,
    attention: latestWeek.attention - baselineScores.attention
  };
  
  return (deltas.regulation + deltas.awareness + deltas.outlook + deltas.attention) / 4;
}
```

### **Calm Rating Average**

```javascript
function calculateAverageCalmRating(dailyLog) {
  const last14Days = dailyLog.slice(-14);
  const ratingsOnly = last14Days
    .map(day => day.calmRating)
    .filter(rating => rating !== null);
  
  return ratingsOnly.reduce((sum, r) => sum + r, 0) / ratingsOnly.length;
}
```

### **Unlock Eligibility Check**

```javascript
async function checkUnlockEligibility(stage) {
  // Load required data
  const dailyLogResult = await window.storage.get('ios:daily_log');
  const dailyLog = JSON.parse(dailyLogResult.value);
  
  const weeklyDeltasResult = await window.storage.get('ios:weekly_deltas');
  const weeklyDeltas = JSON.parse(weeklyDeltasResult.value);
  
  const baselineResult = await window.storage.get('ios:baseline:domain_scores');
  const baselineScores = JSON.parse(baselineResult.value);
  
  // Calculate metrics
  const adherence = calculateAdherence(stage, dailyLog);
  const avgDelta = calculateAverageDelta(weeklyDeltas, baselineScores);
  const avgCalm = calculateAverageCalmRating(dailyLog);
  
  // Define thresholds per stage
  const thresholds = {
    1: { adherence: 80, delta: 0.3, calmRating: 3.0 },
    2: { adherence: 80, delta: 0.5, feltSense: 3.0 },
    3: { adherence: 80, delta: 0.5, identityIntegration: 3.0 }
  };
  
  const threshold = thresholds[stage];
  
  // Check criteria
  const eligible = {
    adherence: adherence >= threshold.adherence,
    delta: avgDelta >= threshold.delta,
    calmRating: stage === 1 ? avgCalm >= threshold.calmRating : true,
    // Add stage-specific checks for Stage 2+
  };
  
  const allMet = Object.values(eligible).every(v => v === true);
  
  // Store result
  await window.storage.set('ios:unlock_status', JSON.stringify({
    currentStage: stage,
    nextStage: stage + 1,
    unlockEligible: allMet,
    criteria: {
      adherence: { required: threshold.adherence, actual: adherence, met: eligible.adherence },
      delta: { required: threshold.delta, actual: avgDelta, met: eligible.delta },
      calmRating: { required: threshold.calmRating, actual: avgCalm, met: eligible.calmRating }
    },
    evaluationDate: new Date().toISOString()
  }));
  
  return allMet;
}
```
